1 A highlight annotation mode for Emacs

One of my students asked about highlighting text in emacs for note-taking. For temporary use, this is pretty easy, you just set a property on a region. For example, we can do this: 

(defun highlight-region (beg end)
 (interactive "r")
 (set-text-properties
  beg end
  '(font-lock-face (:background "Light Salmon")
                   highlighted t)))

This sets the background color, and another property "highlighted". The trouble is this is transient. When I close the file, the highlights are lost. We can save them to a file though, and reload them. As long as we are diligent about that, and updating them, we can store highlighted text, and have them for notes. 

(require 'eyedropper)

(defun highlight-save ()
  (interactive)
  (setq *highlighted-regions* '())
  (eyedropper-background)
  (goto-char (point-min))
  (when (get-text-property (point) 'highlighted)
    (let ((start (point))
          (end (next-single-property-change (point) 'highlighted))
          (color eyedrop-picked-background))
      (add-to-list '*highlighted-regions* (list start end color))))

  ;; now search forward for highlights
  (let (p result)
    (while (setq p (next-single-property-change (point) 'highlighted))
      (goto-char p)
      (setq result (button-lock-find-extent (point) 'highlighted))
      (eyedropper-background)
      (add-to-list '*highlighted-regions* (list (car result)
                                                (cdr result)
                                                eyedrop-picked-background))
      (goto-char (cdr result))))
  (let ((fname (concat (buffer-file-name) ".highlights")))

    (if *highlighted-regions*
        (message-box "Saving %s" *highlighted-regions*)
        (with-temp-file fname
          (print *highlighted-regions* (current-buffer)))
      ;; get rid of file if there are not highlights
      (when (file-exists-p fname)
        (delete-file fname)))))

(add-hook 'after-save-hook 'highlight-save)

Here, we can read the contents and apply the highlights. We set this up on a hook for org-mode, so it will apply them on when we open org-files. 

(defun highlight-load ()
  (interactive)
  (let ((fname (concat (buffer-file-name) ".highlights")))
    (when (file-exists-p fname)
      (mapcar
       (lambda (entry)
         (let ((beg (nth 0 entry))
               (end (nth 1 entry))
               (color (nth 2 entry)))
           (set-text-properties
            beg end
            `(font-lock-face (:background ,color)
                             highlighted t))))
       (with-temp-buffer (insert-file-contents fname)
                         (read (current-buffer)))))))

(add-hook 'org-mode-hook 'highlight-load)

what else can we do? Want to get rid of the highlights? 

(defun highlight-clear ()
  (interactive)
  (let ((fname (concat (buffer-file-name) ".highlights")))
    (when (file-exists-p fname)
      (mapcar
       (lambda (entry)
         (let ((beg (nth 0 entry))
               (end (nth 1 entry)))
           (set-text-properties
            beg end nil)))
       (with-temp-buffer (insert-file-contents fname)
                         (read (current-buffer)))))))

How about list your highlights? 

(defun highlight-list ()
  (interactive)
  (highlight-save)
  (let ((cb (current-buffer))
        (hls (mapcar
              (lambda (entry)
                (list (nth 0 entry)
                      (buffer-substring (nth 0 entry) (nth 1 entry))))
              *highlighted-regions*)))
    (switch-to-buffer-other-window "*highlights*") (org-mode)
    (erase-buffer)
    (dolist (s hls)
      (insert (format "%s [[elisp:(progn (switch-to-buffer-other-window \"%s\")(goto-char %s))][link]]\n" (nth 1 s) cb (nth 0 s))))))
